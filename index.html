      var finalPicture = function(penguin)
    {
        var object = {pic:penguin.picture,
                     finalGrade:penguin.final[0].grade};
        
        return object;
        
    }
    
    var answer=[classroom.map(finalPicture)];

This function first defines an object which has a penguin's picture and the grade it received on the final. Using classroom.map it takes all of penguins in the classroom array and runs the function for each one. Then it takes all of the objects which contain the picture and final grade for their respective penguin and puts them into a new array.
